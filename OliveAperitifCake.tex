\subsection{Adapted from {\it French Food at Home} by Laura Calder}

\vspace{0.3in}
\begin{description}

\item[Ingredients:]\ \\
	\begin{itemize}
	\item 1 1/2 cups flour
	\item 11/2 teaspoon of baking powder
	\item 1/2  teaspoon ground pepper
	\item 3 eggs, lightly beaten
	\item 1/2 cup plus 1 Tbs butter, melted
	\item 2/3 cup olives, pitted and sliced
	\item 1/3 cup of sun-dried tomatoes, chopped
	\item 1 cup Gruyere cheese cut into small cubes (or grated Parmesan)
	\item 1 pinch of salt, to taste
	\item 1 Tbs chopped fresh rosemary (optional)
	\end{itemize}

\item[Procedure:]\ \\
	\begin{itemize}
	\item Heat the oven to 350¼F/180¼C. 
	\item Butter and flour a loaf pan. 
	\item Sift the flour and baking powder into a mixing bowl, and stir the pepper through. 
	\item Beat the eggs into the flour. 
	\item Beat in the melted butter. 
	\item Stir through the remaining ingredients to distribute evenly. 
	\item Spoon the batter into the pan and bake 45 minutes until a toothpick inserted in the centre comes out clean. 
	\item Cool slightly. Remove from pan. Slice, and serve warm.
	\end{itemize}
\end{description}
